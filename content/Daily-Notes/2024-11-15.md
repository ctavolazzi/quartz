import os
import autogen
from autogen import AssistantAgent, UserProxyAgent

llm_config = {"model": "gpt-4", "api_key": os.environ["OPENAI_API_KEY"]}

with autogen.coding.DockerCommandLineCodeExecutor(work_dir="coding") as code_executor:
    assistant = AssistantAgent("assistant", llm_config=llm_config)
    user_proxy = UserProxyAgent(
        "user_proxy", code_execution_config={"executor": code_executor}
    )

    # Start the chat
    user_proxy.initiate_chat(
        assistant,
        message="Develop a text-based Choose Your Own Adventure game using **python only** and nothing else. Run the game, test it, make sure it has the following elements: 1. a text-based UI 2. the ability to choose between three possible options with a fourth option that is just 'other' 3. the game should always start in the same place with the same character named Sam Iker waking up from a dream about flying up around a skyscraper 4. the game should be called 'Teleport Massive CYOA' 5. the game should log the setup and running of the game 6. the game should generate markdown files formatted in obsidian style for each event with a set of choices 7. the game should use the markdown files as reference when generating new scenarios 8. the game should keep a table with an index of scenarios 9. the game should keep a table of characters with their stats and states 10. the game should keep a table of locations with their stats and states 11. the game should keep a table of items with their stats and states 12. the game should update the tables whenever the player makes choices that affect the state of the element in question 13. the game should keep track of the actions the player took in another separate table that tracks player choices 14. the game should generate believable character dialogue using the information generated by playing the game contained in the tables and the markdown scenario files 15. the game should periodically update its own state and have a librarian function that reindexes and keeps track of the state of the game and makes sure everything is consistent 16. the game should not overdo it and should keep everything as simple and straigntforward as possible",
